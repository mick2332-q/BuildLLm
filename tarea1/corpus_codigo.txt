def calculate_fibonacci(n):
    if n <= 1:
        return n
    return calculate_fibonacci(n-1) + calculate_fibonacci(n-2)

def train_neural_network(X_train, y_train, epochs=100, learning_rate=0.001):
    weights = np.random.randn(X_train.shape[1])
    bias = 0.0
    
    for epoch in range(epochs):
        predictions = sigmoid(np.dot(X_train, weights) + bias)
        loss = binary_cross_entropy(y_train, predictions)
        
        dw = np.dot(X_train.T, (predictions - y_train)) / len(y_train)
        db = np.mean(predictions - y_train)
        
        weights -= learning_rate * dw
        bias -= learning_rate * db
        
        if epoch % 10 == 0:
            print(f"Epoch {epoch}, Loss: {loss:.4f}")
    
    return weights, bias

def preprocess_text_data(sentences, max_length=512):
    tokenizer = AutoTokenizer.from_pretrained('bert-base-uncased')
    encoded = tokenizer.batch_encode_plus(
        sentences,
        max_length=max_length,
        padding='max_length',
        truncation=True,
        return_tensors='pt'
    )
    return encoded['input_ids'], encoded['attention_mask']
